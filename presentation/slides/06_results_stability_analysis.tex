\begin{frame}{Experiment and analysis: Stability analysis}
    \begin{columns}[t]
        \begin{column}{0.48\textwidth}
            \textbf{Uniform FP8 limitation}
            
            \small
            Baseline (Uniform E5M2) exhibits instability:
            \begin{itemize}
                \item \textit{Cause}: MLP gradients have narrow distribution
                \item \textit{Effect}: Coarse E5M2 quantization induces noise $\rightarrow$ spikes
            \end{itemize}
        \end{column}

        \begin{column}{0.48\textwidth}
            \textbf{Layer-wise solution}
            
            \small
            Component-adaptive assignment:
            \begin{itemize}
                \item \textbf{Attention}: E5M2 accommodates gradient extrema
                \item \textbf{MLP}: E4M3 ensures precision for stable convergence
            \end{itemize}
        \end{column}
    \end{columns}

    \vspace{0.5cm}
    \centering
    \textbf{Result}: Format-component alignment eliminates training variance spikes
\end{frame}
